# Propagation event
province_event = { 
	id = 357690
	title = "EVTNAME357690" #Ozpox Spreads!
	desc = "EVTDESC357690"
	mean_time_to_happen = {
		months = 6
		modifier = {
			factor = 3.0
			country = { NOT = { has_country_flag = ozpox_outbreak } }
		}
		modifier = { # Proximity to coast, capital, or railroad and high life rating speeds the spread
			factor = 0.5
			OR = { is_coastal = yes
				is_capital = yes
				life_rating = 34
				has_building = railroad
			}
		}
		modifier = { # Low life rating slows the spread
			factor = 2.0
			NOT = {
				OR = {
					life_rating = 31
					is_coastal = yes
					is_capital = yes
				}
			}
		}
		modifier = { # Country border status
			factor = 1.5
			owner = {
				AND = { border_policy = closed_borders
					NOT = { has_country_flag = ozpox_outbreak }
				}
			}
		}
		modifier = { # Country is under quarantine
			factor = 3.0
			owner = { has_country_flag = ozpox_quarantine }
		}
		modifier = { # Neighboring countries' status
			factor = 1.2
			any_neighbor_country = {
				NOT = { has_country_flag = ozpox_outbreak }
				NOT = { has_country_modifier = pandemic_ozpox }
				civilized = yes
			}
		}
		modifier = { # Already many infected provinces slows the spread
			factor = 1.2
			owner = { check_variable = { which = infected_provinces value = 8 } }
		}
		modifier = { # High literacy slows the spread
			factor = 1.2
			owner = { literacy = 60 }
		}
		modifier = { # Continent modifier
			factor = 1.5
			OR = { continent = africa continent = asia }
		}
	}
	trigger = {
		NOT = { has_province_modifier = ozpox_outbreak }
		owner = { NOT = { has_country_flag = ozpox_nation_pandemic } }
		OR = {
			any_neighbor_province = { has_province_modifier = ozpox_outbreak }
			any_neighbor_country = { has_country_modifier = pandemic_ozpox }
			owner = { any_owned_province = { has_province_modifier = ozpox_outbreak } }
		}
	}

	immediate = {
		country = { change_variable = { which = infected_provinces value = 2 } }
	}

	option = {
		name = "EVT357690OPTA"
		add_province_modifier = {
			name = ozpox_outbreak
			duration = 150
		}
		random_list = {
			50 = {
				random_neighbor_province = {
					limit = {
						NOT = { has_province_modifier = ozpox_outbreak }
						owner = { NOT = { has_country_modifier = pandemic_ozpox } }
					}
					add_province_modifier = {
						name = ozpox_outbreak
						duration = 150
					}
				}
			}
			50 = {
				owner = {
					limit = {
						any_owned_province = { NOT = { has_province_modifier = ozpox_outbreak } }
					}
					random_owned = {
						limit = { NOT = { has_province_modifier = ozpox_outbreak } }
						add_province_modifier = {
							name = ozpox_outbreak
							duration = 150
						}
					}
				}
			}
		}
   	}
}
